	#import <Foundation/Foundation.h>

	int sum(int* numbers, int numbersCount)
	{
	  int min = numbers[0];
	  int max = numbers[0];
	  int sum = 0;
	  
	  if (numbersCount < 2) {
	    return 0;
	  }
	  
	  for (int i = 0; i < numbersCount; i++) {
	    if (numbers[i] < min) {
	      min = numbers[i];  
	    }
	    
	    if (numbers[i] > max) {
	      max = numbers[i];
	    }
	    
	    sum = sum + numbers[i];  
	  }
	  
	  return sum - (min + max);
	}
